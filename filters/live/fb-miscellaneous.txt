! friends you may know (news feed)
facebook.com##div[role="feed"] > div div[role="region"] > div > div > div:nth-of-type(1) > div > div > div > div > div > div > div > div > div > span > span > div > a[href*="/friends/suggestions/"][role="link"]:upward(div[role="feed"] > div)

! stories | reels | rooms (news feed)
! - structure must be specific as tablist is used elsewhere. not found under [role="feed"].
facebook.com##div[role="main"] > div > div > div > div > div > div > div > div[role="tablist"]:upward(4)

! Reel/Short Videos (collection) (news feed)
facebook.com##div[role="feed"] > div a[href="/reel/?s=ifu_see_more"]:upward(div[role="feed"] > div)

! Reel post
facebook.com##div[role="feed"] > div div a[href*="/reel/"] video:upward(div[role="feed"] > div)


! "Information box" - climate info
facebook.com##div[role="feed"] > div a[href*="/climatescienceinfo/?hoisted"]:upward(5)
! "Information box" - coronavirus info
facebook.com##div[role="feed"] > div a[href*="/coronavirus_info/"]:upward(5)
! "Information box" - subscribe
facebook.com##div[role="feed"] > div a[href*="/support/"]:upward(5)
