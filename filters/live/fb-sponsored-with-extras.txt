! sponsored (news / groups / videos / search)
! -- clear out hidden elements to enable detection of sponsored posts (news / groups / search)
! --- do not use span[id] - search page doesn't have this. instead, use span[class]
facebook.com##[role="feed"] > div span > span > span > a[href="#"] :matches-css(display: none):remove()
facebook.com##[role="feed"] > div span > span > span > a[href="#"] :matches-css(position: absolute):remove()
facebook.com##[role="feed"] > div span > span > span > a[href*="/ads/"] :matches-css(display: none):remove()
facebook.com##[role="feed"] > div span > span > span > a[href*="/ads/"] :matches-css(position: absolute):remove()
! -- clear out hidden elements to enable detection of sponsore posts (Videos)
facebook.com##[id="watch_feed"] > div span > span > a[href="#"] :matches-css(display: none):remove()
facebook.com##[id="watch_feed"] > div span > span > a[href="#"] :matches-css(position: absolute):remove()
facebook.com##[id="watch_feed"] > div span > span > a[href*="/ads/"] :matches-css(display: none):remove()
facebook.com##[id="watch_feed"] > div span > span > a[href*="/ads/"] :matches-css(position: absolute):remove()
! -- clear out hidden elements to allow some of the rules below to work ("Facebook")
facebook.com##div[role="feed"] > div div[data-0]:remove()
!
! -- hide sponsored posts (news / groups / search feed)
! -- שעה אחת = Hebrew for "one hour"; ഒരു മണിക്കൂർ = Malaylam for "one/an hour"
! -- [\u0660-\u0669] = Arabic digits 
! -- [0-9] = ASCII digits
facebook.com##[role="feed"] > div span > span > span > a[href*="/ads/"] > span > span[class] > [style*="display"]:not(:has-text(/(שעה אחת|ഒരു മണിക്കൂർ|[\u0660-\u0669]|[0-9]|^$)/)):upward([role="feed"] > div)
facebook.com##[role="feed"] > div span > span > span > a[href="#"] > span > span[class] > [style*="display"]:not(:has-text(/(שעה אחת|ഒരു മണിക്കൂർ|[\u0660-\u0669]|[0-9]|^$)/)):upward([role="feed"] > div)
!
! -- non-obfuscated version
facebook.com##[role="feed"] > div span > span > span > a[href*="/ads/"] > span > span:not([style]) > *:not(:has-text(/(שעה אחת|ഒരു മണിക്കൂർ|[\u0660-\u0669]|[0-9]|^$)/)):upward([role="feed"] > div)
facebook.com##[role="feed"] > div span > span > span > a[href="#"] > span > span:not([style]) > *:not(:has-text(/(שעה אחת|ഒരു മണിക്കൂർ|[\u0660-\u0669]|[0-9]|^$)/)):upward([role="feed"] > div)
!
!
! -- hide sponsored posts (watch videos feed)
facebook.com##div[id="watch_feed"] > div span > span > span > a[href="#"] > span > span[class] > [style*="display"]:not(:has-text(/(שעה אחת|ഒരു മണിക്കൂർ|[\u0660-\u0669]|[0-9]|^$)/)):upward(div#watch_feed > div > div > div > div > div > div[class])
! -- non-obfuscated version
facebook.com##div[id="watch_feed"] > div span > span > span > a[href="#"] > span > span:not([style]) > *:not(:has-text(/(שעה אחת|ഒരു മണിക്കൂർ|[\u0660-\u0669]|[0-9]|^$)/)):upward(div#watch_feed > div > div > div > div > div > div[class])
!
!
! paid partnership posts (news feed)
facebook.com##div[role="feed"] > div span[dir] > span[id] a[href^="/business/help/"]:upward(div[role="feed"] > div)
!
! sponsored * paid for by ...... (news feed)
facebook.com##div[role="feed"] > div div:nth-child(2) > div > div:nth-child(2) > span[class] > span[id] > div:nth-child(2):not(>*):upward(div[role="feed"]> div)
!
!
! hide sponsored posts (marketplace feed)
! - nb: AdGuard Base filter list hides the label only - not the marketplace item. here, we do both.
! - nb: "Sponsored" is not obfuscated
facebook.com##a[href="/ads/about/?entry_product=ad_preferences"]:upward(1)
facebook.com##div[class] > span > div > object > a[href*="/ads/about/"]:upward(div[style*="max-width"])
! - nb: fb dropped the sponsored details from the item, so use another method to detect the sponsored item (Sept 2022)
facebook.com##div[class] > span > div a:not([href*="marketplace"]):upward(div[style*="max-width"])
!

! suggested (news feed)
! - last span needs :not(>*) as it doesn't have any child elements. Without it, it'll cause false-positives.
facebook.com##div[role="feed"] div > div > div > div > div > div > div > div > div > div > div > div > div:nth-of-type(2) > div > div:nth-of-type(1) > div > div > div > div > span:not(>*):upward([role="feed"] > div)
!
! suggested (groups feed)
facebook.com##div[role="article"][aria-posinset] > div > div > div > div > div > div > div > div:first-of-type:has(i[data-visualcompletion="css-img"][style]):upward(div[role="feed"] > div)
!
! suggested - top of feed / other column (groups feed)
facebook.com##a[href*="/groups/discover"] > span > span:upward(21)

! friends you may know (news feed)
facebook.com##div[role="feed"] > div div[role="region"] > div > div > div:nth-of-type(1) > div > div > div > div > div > div > div > div > div > span > span > div > a[href*="/friends/suggestions/"][role="link"]:upward(div[role="feed"] > div)
!
! stories | reels | rooms (news feed)
! - structure must be specific as tablist is used elsewhere. not found under [role="feed"].
facebook.com##div[role="main"] > div > div > div > div > div > div > div > div[role="tablist"]:upward(4)
