! sponsored (news / groups / videos / search)
! -- clear out hidden elements to enable detection of sponsored posts (news / groups / search)
! --- do not use span[id] - search page doesn't have this. instead, use span[class]
facebook.com##[role="feed"] > div span > span > span > a[href="#"] :matches-css(display: none):remove()
facebook.com##[role="feed"] > div span > span > span > a[href="#"] :matches-css(position: absolute):remove()
facebook.com##[role="feed"] > div span > span > span > a[href*="/ads/"] :matches-css(display: none):remove()
facebook.com##[role="feed"] > div span > span > span > a[href*="/ads/"] :matches-css(position: absolute):remove()
! -- clear out hidden elements to enable detection of sponsore posts (Videos)
facebook.com##[id="watch_feed"] > div span > span > a[href="#"] :matches-css(display: none):remove()
facebook.com##[id="watch_feed"] > div span > span > a[href="#"] :matches-css(position: absolute):remove()
facebook.com##[id="watch_feed"] > div span > span > a[href*="/ads/"] :matches-css(display: none):remove()
facebook.com##[id="watch_feed"] > div span > span > a[href*="/ads/"] :matches-css(position: absolute):remove()
! -- clear out hidden elements to allow some of the rules below to work ("Facebook")
facebook.com##div[role="feed"] > div div[data-0]:remove()

! -- hide sponsored posts (news / groups / search feed)
! -- שעה אחת = Hebrew for "one hour"
! -- [\u0660-\u0669] = Arabic digits 
! -- [0-9] = ASCII digits
facebook.com##[role="feed"] > div span > span > span > a[href*="/ads/"] > span > span[class] > [style*="display"]:not(:has-text(/(שעה אחת|[\u0660-\u0669]|[0-9]|^$)/)):upward([role="feed"] > div):style(border:6px dotted brown !important; padding: 0 0.25rem; width:66% !important;)
facebook.com##[role="feed"] > div span > span > span > a[href="#"] > span > span[class] > [style*="display"]:not(:has-text(/(שעה אחת|[\u0660-\u0669]|[0-9]|^$)/)):upward([role="feed"] > div):style(border:6px dotted red !important; padding: 0 0.25rem; width:66% !important;)

! -- non-obfuscated version
facebook.com##[role="feed"] > div span > span > span > a[href*="/ads/"] > span > span:not([style]) > *:not(:has-text(/(שעה אחת|[\u0660-\u0669]|[0-9]|^$)/)):upward([role="feed"] > div):style(border:6px dotted maroon !important; padding: 0 0.25rem; width:66% !important;)
facebook.com##[role="feed"] > div span > span > span > a[href="#"] > span > span:not([style]) > *:not(:has-text(/(שעה אחת|[\u0660-\u0669]|[0-9]|^$)/)):upward([role="feed"] > div):style(border:6px dotted firebrick !important; padding: 0 0.25rem; width:66% !important;)


! -- hide sponsored posts (watch videos feed)
facebook.com##div[id="watch_feed"] > div span > span > span > a[href="#"] > span > span[class] > [style*="display"]:not(:has-text(/(שעה אחת|[\u0660-\u0669]|[0-9]|^$)/)):upward(div#watch_feed > div > div > div > div > div > div[class]):style(border:6px dotted darkred !important; padding: 0 0.25rem; width:66% !important;)
! -- non-obfuscated version
facebook.com##div[id="watch_feed"] > div span > span > span > a[href="#"] > span > span:not([style]) > *:not(:has-text(/(שעה אחת|[\u0660-\u0669]|[0-9]|^$)/)):upward(div#watch_feed > div > div > div > div > div > div[class]):style(border:6px dotted gold !important; padding: 0 0.25rem; width:66% !important;)


! paid partnership posts (news feed)
facebook.com##div[role="feed"] > div span[dir] > span[id] a[href^="/business/help/"]:upward(div[role="feed"] > div):style(border:6px dotted blue !important; padding: 0 0.25rem; width:66% !important;)


! marketplace feed
! - nb: AdGuard Base filter list hides the label only - not the marketplace item. here, we do both.
! - nb: "Sponsored" is not obfuscated
facebook.com##a[href="/ads/about/?entry_product=ad_preferences"]:upward(1):style(border:6px dotted purple !important; padding: 0 0.25rem; width:66% !important;)
facebook.com##div[class] > span > div > object > a[href*="/ads/about/"]:upward(div[style*="max-width"]):style(border:6px dotted purple !important; padding: 0 0.25rem; width:66% !important;)
