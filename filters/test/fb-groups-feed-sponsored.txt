! sponsored (groups / search)
! -- clear out hidden elements to enable detection of sponsored posts (groups / search)
! --- do not use span[id] - search page doesn't have this. instead, use span[class]
facebook.com##[role="feed"] > div span > span > span > a[href="#"] :matches-css(display: none):remove()
facebook.com##[role="feed"] > div span > span > span > a[href="#"] :matches-css(position: absolute):remove()
facebook.com##[role="feed"] > div span > span > span > a[href*="/ads/"] :matches-css(display: none):remove()
facebook.com##[role="feed"] > div span > span > span > a[href*="/ads/"] :matches-css(position: absolute):remove()
! -- clear out hidden elements to allow some of the rules below to work ("Facebook")
facebook.com##div[role="feed"] > div div[data-0]:remove()
!
! -- hide sponsored posts (groups / search feed)
! -- שעה אחת = Hebrew for "one hour"; ഒരു മണിക്കൂർ = Malaylam for "one/an hour"
! -- [\u0660-\u0669] = Arabic digits 
! -- [0-9] = ASCII digits
facebook.com##[role="feed"] > div span > span > span > a[href*="/ads/"] > span > span[class] > [style*="display"]:not(:has-text(/(שעה אחת|ഒരു മണിക്കൂർ|[\u0660-\u0669]|[0-9]|^$)/)):upward([role="feed"] > div):style(border:6px dotted brown !important; padding: 0 0.25rem; width:66% !important;)
facebook.com##[role="feed"] > div span > span > span > a[href="#"] > span > span[class] > [style*="display"]:not(:has-text(/(שעה אחת|ഒരു മണിക്കൂർ|[\u0660-\u0669]|[0-9]|^$)/)):upward([role="feed"] > div):style(border:6px dotted red !important; padding: 0 0.25rem; width:66% !important;)
!
! -- non-obfuscated version
facebook.com##[role="feed"] > div span > span > span > a[href*="/ads/"] > span > span:not([style]) > *:not(:has-text(/(שעה אחת|ഒരു മണിക്കൂർ|[\u0660-\u0669]|[0-9]|^$)/)):upward([role="feed"] > div):style(border:6px dotted maroon !important; padding: 0 0.25rem; width:66% !important;)
facebook.com##[role="feed"] > div span > span > span > a[href="#"] > span > span:not([style]) > *:not(:has-text(/(שעה אחת|ഒരു മണിക്കൂർ|[\u0660-\u0669]|[0-9]|^$)/)):upward([role="feed"] > div):style(border:6px dotted firebrick !important; padding: 0 0.25rem; width:66% !important;)
